def plotDot(point):
    folium.CircleMarker(location=[point.latitude, point.longitude]
                       ,radius=4
                       ,weight=2
                       # input text or visualization for object displayed when clicking.
                       # information you want to show when click on the marker
                       ,popup=#[point.Country, point.TotalConfirmed, point.TotalRecovered]
                        folium.Popup(str([point.Country, point.TotalConfirmed, point.TotalRecovered]))
                       ,fill_color='00000' #,00000 tranparent, #3186cc,BuPu,YlGn
                       ).add_to(m)
